Quick Start Example

// Import registry and components import { ComponentRegistry } from './modules/registry.js'; import { Modal, Toast } from './modules/components/overlays.js'; // Register components ComponentRegistry.register(Modal).register(Toast); // Create a modal instance const modal = ComponentRegistry.create('modal', { title: 'Hello World', content: 'This is a reusable component!', actions: [ { id: 'close', label: 'Close', callback: () => console.log('Closed') } ] }); // Mount to DOM document.body.appendChild(modal.element); modal.mount(); // Update props modal.update({ title: 'Updated Title' }); // Cleanup modal.destroy();
